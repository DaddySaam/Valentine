    <script>
        const mainScreen = document.getElementById('main-screen');
        const yesScreen = document.getElementById('yes-screen');
        const yesButton = document.getElementById('yes-button');
        const noButton = document.getElementById('no-button');
        const body = document.body;
        const music = document.getElementById('romantic-music');

        // Function to move NO button to a random position
        function moveNoButton() {
            const maxX = window.innerWidth - noButton.offsetWidth;
            const maxY = window.innerHeight - noButton.offsetHeight;
            const randomX = Math.random() * maxX;
            const randomY = Math.random() * maxY;
            noButton.style.left = randomX + 'px';
            noButton.style.top = randomY + 'px';
            body.style.backgroundColor = '#ffb3d9'; // Turn background pink
        }

        // Event listener for NO button
        noButton.addEventListener('click', moveNoButton);

        // Event listener for YES button
        yesButton.addEventListener('click', () => {
            mainScreen.style.display = 'none';
            yesScreen.style.display = 'flex';
            music.play(); // Start romantic music
        });
